## 1. Auto Consultation Summary

### Overview

After each consultation session ends, the doctor sees a draft summary automatically generated by Claude based on the consultation notes they enter. The doctor can edit and approve it with one click. The approved summary becomes part of the patient's medical record.

### Prisma Schema Additions

```prisma
model ConsultationSummary {
  id             String      @id @default(cuid())
  appointmentId  String      @unique
  appointment    Appointment @relation(fields: [appointmentId], references: [id])
  draftContent   String      // AI-generated draft
  finalContent   String?     // Doctor-approved final version
  status         SummaryStatus @default(DRAFT)
  generatedAt    DateTime    @default(now())
  approvedAt     DateTime?
  approvedBy     String?     // Doctor userId
  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt
}

enum SummaryStatus {
  DRAFT
  APPROVED
}
```

Add to your existing `Appointment` model:

```prisma
consultationNotes  String?
summary            ConsultationSummary?
```

### NestJS Backend

**Module:** `consultation-summary`

**POST `/consultation-summary/generate/:appointmentId`**

```typescript
// consultation-summary.service.ts
async generateSummary(appointmentId: string): Promise<ConsultationSummary> {
  const appointment = await this.prisma.appointment.findUnique({
    where: { id: appointmentId },
    include: {
      patient: { include: { profile: true } },
      doctor: { include: { profile: true } },
    },
  });

  if (!appointment) throw new NotFoundException('Appointment not found');

  const prompt = `
You are a medical documentation assistant. Based on the following consultation data, generate a concise and professional medical consultation summary.

Patient: ${appointment.patient.profile.firstName} ${appointment.patient.profile.lastName}
Age: ${this.calculateAge(appointment.patient.profile.birthday)}
Doctor: Dr. ${appointment.doctor.profile.lastName} (${appointment.doctor.profile.specialization})
Date: ${appointment.scheduledAt.toDateString()}
Duration: ${appointment.durationMinutes} minutes

Doctor's Notes:
${appointment.consultationNotes || 'No notes provided.'}

Generate a structured summary with the following sections:
- Chief Complaint
- Assessment
- Plan / Recommendations
- Follow-up Instructions (if any)

Keep the language professional, concise, and clinically appropriate. Do not fabricate medical details not present in the notes.
  `.trim();

  const response = await this.anthropic.messages.create({
    model: 'claude-sonnet-4-20250514',
    max_tokens: 1000,
    messages: [{ role: 'user', content: prompt }],
  });

  const draftContent = response.content[0].type === 'text'
    ? response.content[0].text
    : '';

  return this.prisma.consultationSummary.upsert({
    where: { appointmentId },
    create: { appointmentId, draftContent },
    update: { draftContent, status: 'DRAFT' },
  });
}

async approveSummary(
  appointmentId: string,
  finalContent: string,
  doctorId: string,
): Promise<ConsultationSummary> {
  return this.prisma.consultationSummary.update({
    where: { appointmentId },
    data: {
      finalContent,
      status: 'APPROVED',
      approvedAt: new Date(),
      approvedBy: doctorId,
    },
  });
}
```

**POST `/consultation-summary/approve/:appointmentId`**

- Body: `{ finalContent: string }`
- Auth: Doctor only

### Next.js Frontend

**Page:** `/dashboard/doctor/appointments/[id]/summary`

```tsx
// SummaryEditor component (Doctor view)
'use client';

export function SummaryEditor({ appointmentId }: { appointmentId: string }) {
  const [content, setContent] = useState('');
  const [isGenerating, setIsGenerating] = useState(false);
  const [isApproved, setIsApproved] = useState(false);

  const handleGenerate = async () => {
    setIsGenerating(true);
    const res = await fetch(`/api/consultation-summary/generate/${appointmentId}`, {
      method: 'POST',
    });
    const data = await res.json();
    setContent(data.draftContent);
    setIsGenerating(false);
  };

  const handleApprove = async () => {
    await fetch(`/api/consultation-summary/approve/${appointmentId}`, {
      method: 'POST',
      body: JSON.stringify({ finalContent: content }),
    });
    setIsApproved(true);
  };

  return (
    <div>
      <div className="flex items-center justify-between mb-4">
        <h2 className="text-lg font-semibold">Consultation Summary</h2>
        <button onClick={handleGenerate} disabled={isGenerating}>
          {isGenerating ? 'Generating...' : '✨ Generate with AI'}
        </button>
      </div>
      <textarea
        value={content}
        onChange={(e) => setContent(e.target.value)}
        rows={12}
        className="w-full border rounded-lg p-4 font-mono text-sm"
        placeholder="AI-generated summary will appear here. You can edit before approving."
      />
      <div className="flex justify-end mt-3">
        <button
          onClick={handleApprove}
          disabled={!content || isApproved}
          className="bg-green-600 text-white px-6 py-2 rounded-lg"
        >
          {isApproved ? '✓ Approved' : 'Approve & Save'}
        </button>
      </div>
    </div>
  );
}
```

## 2. Follow-up Care Reminders

### Overview

After a consultation, the doctor can set a follow-up date and optionally add care instructions (medications, lifestyle, recheck). The patient receives a push notification and an in-app reminder on the follow-up date.

### Prisma Schema Additions

```prisma
model FollowUpReminder {
  id              String      @id @default(cuid())
  appointmentId   String
  appointment     Appointment @relation(fields: [appointmentId], references: [id])
  patientId       String
  patient         User        @relation("PatientReminders", fields: [patientId], references: [id])
  followUpDate    DateTime
  instructions    String?
  isAcknowledged  Boolean     @default(false)
  notifiedAt      DateTime?
  createdAt       DateTime    @default(now())
  updatedAt       DateTime    @updatedAt
}
```

### NestJS Backend

**Module:** `follow-up`

**POST `/follow-up`**

```typescript
// follow-up.service.ts
async createFollowUp(dto: CreateFollowUpDto, doctorId: string) {
  const appointment = await this.prisma.appointment.findUnique({
    where: { id: dto.appointmentId },
    include: { patient: true },
  });

  if (appointment.doctor.id !== doctorId) {
    throw new ForbiddenException();
  }

  return this.prisma.followUpReminder.create({
    data: {
      appointmentId: dto.appointmentId,
      patientId: appointment.patientId,
      followUpDate: new Date(dto.followUpDate),
      instructions: dto.instructions,
    },
  });
}
```

**Notification Scheduler** — use `@nestjs/schedule`:

```typescript
// follow-up.scheduler.ts
@Injectable()
export class FollowUpScheduler {
  constructor(
    private prisma: PrismaService,
    private notificationsService: NotificationsService,
  ) {}

  @Cron(CronExpression.EVERY_DAY_AT_8AM)
  async sendDueReminders() {
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    const tomorrow = new Date(today);
    tomorrow.setDate(tomorrow.getDate() + 1);

    const dueReminders = await this.prisma.followUpReminder.findMany({
      where: {
        followUpDate: { gte: today, lt: tomorrow },
        notifiedAt: null,
      },
      include: { patient: true, appointment: { include: { doctor: true } } },
    });

    for (const reminder of dueReminders) {
      await this.notificationsService.sendPush(reminder.patientId, {
        title: 'Follow-up Reminder',
        body: `Dr. ${reminder.appointment.doctor.profile.lastName} recommends a follow-up today.`,
        data: { type: 'FOLLOW_UP', reminderId: reminder.id },
      });

      await this.prisma.followUpReminder.update({
        where: { id: reminder.id },
        data: { notifiedAt: new Date() },
      });
    }
  }
}
```

**GET `/follow-up/patient/upcoming`** — returns all active reminders for the logged-in patient.

## 3. Async Follow-up Messaging

### Overview

After a consultation, patients can send a short message to the doctor without booking a new appointment. The doctor can reply. This is not a real-time chat — it is asynchronous, scoped to a specific consultation, and limited to post-consultation follow-up questions.

### Prisma Schema Additions

```prisma
model FollowUpMessage {
  id            String      @id @default(cuid())
  appointmentId String
  appointment   Appointment @relation(fields: [appointmentId], references: [id])
  senderId      String
  sender        User        @relation(fields: [senderId], references: [id])
  content       String
  isRead        Boolean     @default(false)
  createdAt     DateTime    @default(now())
}
```

### NestJS Backend

**Module:** `messages`

**POST `/messages`**

```typescript
async sendMessage(dto: SendMessageDto, senderId: string) {
  const appointment = await this.prisma.appointment.findUnique({
    where: { id: dto.appointmentId },
  });

  // Only patient or assigned doctor can message
  const isAuthorized =
    appointment.patientId === senderId || appointment.doctorId === senderId;
  if (!isAuthorized) throw new ForbiddenException();

  // Only allow messaging on completed appointments
  if (appointment.status !== 'COMPLETED') {
    throw new BadRequestException('Messaging is only available after a consultation.');
  }

  const message = await this.prisma.followUpMessage.create({
    data: {
      appointmentId: dto.appointmentId,
      senderId,
      content: dto.content,
    },
    include: { sender: true },
  });

  // Notify the other party
  const recipientId =
    senderId === appointment.patientId
      ? appointment.doctorId
      : appointment.patientId;

  await this.notificationsService.sendPush(recipientId, {
    title: 'New follow-up message',
    body: dto.content.substring(0, 80),
    data: { type: 'MESSAGE', appointmentId: dto.appointmentId },
  });

  return message;
}
```

**GET `/messages/:appointmentId`** — returns message thread for a consultation (auth: patient or doctor of that appointment only).

**PATCH `/messages/:appointmentId/read`** — marks all messages as read for the current user.

### Next.js Frontend

**Component:** `<FollowUpThread appointmentId={id} currentUserId={user.id} />`

## 4. Doctor Daily Dashboard

### Overview

The first thing doctors see after logging in is a focused daily view: today's appointments in order, a quick patient summary for each, and fast access to records before a session starts.

### NestJS Backend

**GET `/doctor/dashboard/today`**

```typescript
async getTodayDashboard(doctorId: string) {
  const start = new Date();
  start.setHours(0, 0, 0, 0);
  const end = new Date();
  end.setHours(23, 59, 59, 999);

  const appointments = await this.prisma.appointment.findMany({
    where: {
      doctorId,
      scheduledAt: { gte: start, lte: end },
    },
    orderBy: { scheduledAt: 'asc' },
    include: {
      patient: {
        include: {
          profile: true,
          medicalHistory: true,
        },
      },
      followUpMessages: {
        where: { isRead: false },
      },
      summary: true,
    },
  });

  return appointments.map((appt) => ({
    id: appt.id,
    scheduledAt: appt.scheduledAt,
    status: appt.status,
    patient: {
      id: appt.patient.id,
      name: `${appt.patient.profile.firstName} ${appt.patient.profile.lastName}`,
      age: this.calculateAge(appt.patient.profile.birthday),
      profilePicture: appt.patient.profile.profilePicture,
      basicHistory: appt.patient.medicalHistory,
    },
    unreadMessages: appt.followUpMessages.length,
    hasSummary: !!appt.summary?.finalContent,
  }));
}
```

## Shared Utilities

### Notifications Service

All four features use push notifications. Implement a single `NotificationsService` in NestJS:

```typescript
@Injectable()
export class NotificationsService {
  async sendPush(userId: string, payload: {
    title: string;
    body: string;
    data?: Record<string, string>;
  }) {
    // Retrieve the user's FCM token from the database
    const token = await this.getUserFcmToken(userId);
    if (!token) return;
    // Send via Firebase Admin SDK or your push provider
    await this.fcm.send({ token, notification: payload, data: payload.data });
  }
}
```

Store FCM tokens in your `User` model:

```prisma
model User {
  // ...existing fields
  fcmToken  String?
}
```

Register/update token on the frontend when the app loads:

```typescript
// Next.js: call on login
POST /users/fcm-token  { token: string }
```
